#include<iostream>
using namespace std;
long long Catalan(int n);
int main() {
	int num;
	while (cin >> num && num > 0)
	{
		cout << num << ": " << Catalan(num) << endl;
	}
	return 0;
}
long long Catalan(int n) {
	if (n <= 1)
		return 1;

	long long *h = new long long[n + 1];
	h[0] = h[1] = 1;
	for (int i = 2; i <= n; i++)
	{
		h[i] = 0ll;
		for (int j = 0; j < i; j++) //根据递归式计算 h(i)= h(0)*h(i-1)+h(1)*h(i-2) + ... + h(i-1)h(0)  
			h[i] += (h[j] * h[i - 1 - j]);
	}
	long long result = h[n];
	delete[]h;
	return result;
}
